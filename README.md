## Install

```
    npm install tdk-all-api --save
```

## Usage

```
const getWord=require('tdk-all-api');

  const results = async (searching) => {
    try {
      const sonuc = await getWord(searching);
      console.log(sonuc);
    } catch (e) {
      console.error(e);
    }
  };
  results("yazılım")

```
